    # File lib/doro/progressbar.rb, line 29
def start
  Signal.trap("INT") { @interrupt = true }

  while (@progress <= @max_progress && @interrupt == false )
    render_progress
    @progress += 1
    yield
    sleep 1
  end

  print("\r")

  display_notification
end
